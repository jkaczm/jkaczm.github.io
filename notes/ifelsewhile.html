Problem statement: Write a program in that prompts the user to enter an integer n, and prints the numbers from one to n. (1,2,3,...,n)

Solution:


            #include<iostream>
    
            using namespace std;
    
            int main(){
                int n;
                cout << "Enter a value for n: ";
                cin >> n;

                int count = 1;

                while (count <= n){
                    cout << count << " ";
                    count++;
                }

                cout << endl;

                return 0;
            }

Problem statement: modify the above program to print the sum of the numbers from one to n.

Solution:


            #include<iostream>
    
            using namespace std;
    
            int main(){
                int n;
                cout << "Enter a value for n: ";
                cin >> n;

                int count = 1;
                int sum = 0;

                while (count <= n){
                    sum = sum + count;
                    count++;
                }

                cout << "The sum from 1 to " << n << " is " << sum << endl;

                return 0;
            }

Problem statement:Write a program in which the user is prompted to enter a number n, and the program prints all the even numbers from one to n.

Solution:

                return 0;
            }

Problem statement:Write a program in which the user is prompted to enter a number n, and the program prints all the divisors of n.

Solution:


            #include<iostream>
    
            using namespace std;
    
            int main(){
                int n;
                cout << "Enter a value for n: ";
                cin >> n;

                int x = 1;

                while (x <= n){
                    if(n % x == 0){
                        cout << x << " ";
                    }
                    x++;
                }

                cout << endl;

                return 0;
            }
