            #include<iostream>
            
            using namespace std;

            int main(){
                int age;

                cout << "Enter your age: ";
                cin >> age;

                //if age is less than one print "I don't believe you"
                if (age < 1)
                    cout << "I don't believe you" << endl;
                
                //if age is between 20 and 25, print "We're almost the same age"
                if (age >= 20 && age <= 25)
                    cout << "We're almost the same age!" << endl;

                //if age is an odd number
                if (age % 2 == 1)
                    cout << "That's an odd age" << endl;

                return 0;
            }
