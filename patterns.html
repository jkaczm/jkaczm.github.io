Printing rectangles

The following program prints a 5x5 grid of X's


            #include<iostream>
            
            using namespace std;

            int main(){
                int numRows = 5;
                int numCols = 5;

                for(int r = 1; r < numRows; r++){
                    for(int c = 1; c < numCols; c++){
                        cout << "X";
                    }
                    cout << endl;
                }

                return 0;
            }

Output:


            XXXXX
            XXXXX
            XXXXX
            XXXXX
            XXXXX

The following program prints a 5x5 grid of X's and O's where X's are printed on odd numbered rows, and O's are printed on even numbered rows.


            #include<iostream>
            
            using namespace std;

            int main(){
                int numRows = 5;
                int numCols = 5;

                for(int r = 1; r < numRows; r++){
                    for(int c = 1; c < numCols; c++){
                        
                        if(r % 2 == 1){
                            cout << "X";
                        } else {
                            cout << "O";
                        }

                    }
                    cout << endl;
                }
                
                return 0;
            }

Output:


            XXXXX
            OOOOO
            XXXXX
            OOOOO
            XXXXX

The following program prints a plus sign of O's inside a 5x5 grid of X's (the middle rows and middle columns are O's)


            #include<iostream>
            
            using namespace std;

            int main(){
                int numRows = 5;
                int numCols = 5;

                for(int r = 1; r < numRows; r++){
                    for(int c = 1; c < numCols; c++){
